Mapping the reads to the genome index using hisat2

Make a loop for both hisat2 and samtools

      ```bash
#!/usr/bin/env bash

# ----------------------------
# Specify index
# ----------------------------

mapping_index="hisat2index_feywei"

# ----------------------------
# Create output directories
# ----------------------------

mkdir -p feywei_bam_dir
mkdir -p feywei_sam_dir

# ----------------------------
# Process each sample directory
# ----------------------------

for i in *day*/; do
    i=${i%/}   # remove trailing slash

    echo "$i"
    cd "$i" || exit 1

    fwd_input_file="${i}_1.fq.gz"
    rev_input_file="${i}_2.fq.gz"

    echo "$fwd_input_file"
    echo "$rev_input_file"

    sam_file="${i}.sam"
    bam_file="${i}.bam"

    echo "$sam_file"

    # ----------------------------
    # Mapping with HISAT2
    # ----------------------------

    hisat2 -p 4 --dta --rna-strandness RF \
        -x "../${mapping_index}" \
        -1 "$fwd_input_file" \
        -2 "$rev_input_file" \
        -S "$sam_file"

    # ----------------------------
    # Convert SAM → BAM
    # ----------------------------

    ~/storage/software/samtools-1.9/samtools sort \
        -@ 4 \
        -o "$bam_file" \
        "$sam_file"

    # ----------------------------
    # Move outputs
    # ----------------------------

    mv "$sam_file" ../feywei_sam_dir/
    mv "$bam_file" ../feywei_bam_dir/

    cd ..
done
```

```python
import csv

with open("Polca_overlapping.csv", "r") as f, open("longer_genes.csv", "w", newline="") as out_file:
    writer = csv.writer(out_file)

    # Write header
    writer.writerow([
        "gene_id1",
        "gene_id2",
        "gene_len_H",
        "gene_len_A",
        "overlap_len",
        "overlap_rate",
    ])

    # Skip header
    next(f)

    for line in f:
        fields = line.strip().split(",")

        gene_id1 = fields[9]
        gene_id2 = fields[13]

        gene_len_H = int(fields[2]) - int(fields[1]) + 1
        gene_len_A = int(fields[12]) - int(fields[11]) + 1

        overlap_len = int(fields[20])
        overlap_rate = float(fields[22])

        if overlap_rate == 0:
            writer.writerow([
                gene_id1,
                gene_id2,
                gene_len_H,
                gene_len_A - 1,
                overlap_len,
                overlap_rate,
            ])
        else:
            if gene_len_H >= gene_len_A:
                writer.writerow([
                    gene_id1,
                    "NA",
                    gene_len_H,
                    "",
                    overlap_len,
                    overlap_rate,
                ])
            else:
                writer.writerow([
                    gene_id1,
                    gene_id2,
                    "NA",
                    gene_len_A,
                    overlap_len,
                    overlap_rate,
                ])
```
