```r
library(tidyverse)
library(GenomicRanges)

# Add function for conditional mutation
mutate_cond <- function(.data, condition, ..., envir = parent.frame()) {
    condition <- eval(substitute(condition), .data, envir)
    .data[condition, ] <- .data[condition, ] %>% mutate(...)
    .data
}

# Read collinearity file
col_mc <- read.table("zh.collinearity", sep = "\t")
names(col_mc) <- c("ID", "group1", "group2", "eval")

# Remove whitespace from column 1
col_mc$ID <- gsub(" ", "", col_mc$ID)

# Separate collinear block IDs from gene numbering
col_mc <- separate(col_mc, ID, c("blockID", "gene_nr"), sep = "-")

# Read annotation file
anno <- read.table("zh.gff", sep = "\t")
names(anno) <- c("chr", "geneID", "start", "stop")

# Add position information to collinearity file
col <- left_join(col_mc, anno, by = c("group1" = "geneID")) %>%
    select(blockID, gene_nr, group1, chr, start, stop, group2, eval)

names(col) <- c(
    "blockID",
    "gene_nr",
    "group1",
    "group1_chr",
    "group1_start",
    "group1_stop",
    "group2",
    "eval"
)

col <- left_join(col, anno, by = c("group2" = "geneID")) %>%
    select(
        blockID,
        gene_nr,
        group1,
        group1_chr,
        group1_start,
        group1_stop,
        group2,
        chr,
        start,
        stop,
        eval
    )

names(col) <- c(
    "blockID",
    "gene_nr",
    "group1",
    "group1_chr",
    "group1_start",
    "group1_stop",
    "group2",
    "group2_chr",
    "group2_start",
    "group2_stop",
    "eval"
)

col$gene_nr <- gsub(":", "", col$gene_nr)
col <- transform(col, gene_nr = as.integer(gene_nr))

# Function to get most frequent value per group
calculate_mode <- function(x) {
    uniqx <- unique(na.omit(x))
    uniqx[which.max(tabulate(match(x, uniqx)))]
}

# Summarize by blockID
col_blocks <- col %>%
    group_by(blockID) %>%
    summarise(
        size = max(gene_nr),
        chrA = calculate_mode(group1_chr),
        startA = min(group1_start),
        stopA = max(group1_stop),
        chrB = calculate_mode(group2_chr),
        startB = min(group2_start),
        stopB = max(group2_stop),
        .groups = "drop"
    )

# Circos-compatible format
col_circos <- select(col_blocks, -c("blockID", "size"))

write.table(
    col_circos,
    "collinear_blocks_zurich.tbl",
    sep = "\t",
    quote = FALSE,
    row.names = FALSE,
    col.names = FALSE
)
```

  ```r
parse_node_labels <- function(tree) {
    if (is.null(tree$node.label)) {
        warning("No node labels found in the tree.")
        return(tree)
    }

    node_labels <- tree$node.label
    bootstrap_values <- rep(NA, length(node_labels))

    for (i in seq_along(node_labels)) {
        if (!is.na(node_labels[i])) {
            parts <- strsplit(node_labels[i], "/")[[1]]

            if (length(parts) > 1) {
                bootstrap_values[i] <- as.numeric(parts[2])
            }
        }
    }

    tree$node.label <- bootstrap_values
    return(tree)
}
```

```python
# ======================================================
# Extract longest mRNA per gene from FASTA
# ======================================================

def read_fasta(file_path):
    """Read FASTA file into a dictionary: {header: sequence}."""
    sequences = {}

    with open(file_path, "r") as file:
        header = None
        seq = []

        for line in file:
            line = line.strip()

            if line.startswith(">"):
                if header is not None:
                    sequences[header] = "".join(seq)

                header = line
                seq = []
            else:
                seq.append(line)

        if header is not None:
            sequences[header] = "".join(seq)

    return sequences


def find_longest_mrna(sequences):
    """Keep only the longest transcript per gene."""
    longest_mrna = {}

    for header, seq in sequences.items():
        # Extract gene ID from header
        parts = header.split()

        if len(parts) < 2:
            continue

        gene_id = parts[1]
        gene_id = gene_id[5:]  # remove "mRNA" prefix if present

        if gene_id not in longest_mrna:
            longest_mrna[gene_id] = (header, seq)
        else:
            _, longest_seq = longest_mrna[gene_id]

            if len(seq) > len(longest_seq):
                longest_mrna[gene_id] = (header, seq)

    return longest_mrna


def write_fasta(output_path, longest_mrna):
    """Write sequences to FASTA file."""
    with open(output_path, "w") as file:
        for header, seq in longest_mrna.values():
            file.write(f"{header}\n")
            file.write(f"{seq}\n")


# ======================================================
# Main
# ======================================================

if __name__ == "__main__":
    input_file = "feywei_sex_genes_appended_protein.fasta"
    output_file = "longest_mRNAs_protein.fasta"

    sequences = read_fasta(input_file)
    longest_mrna = find_longest_mrna(sequences)

    write_fasta(output_file, longest_mrna)

    print(f"Longest mRNAs written to: {output_file}")
```

   ```python
# ======================================================
# Select longest isoform per gene from FASTA
# ======================================================

def read_fasta(file_path):
    """Read FASTA into dictionary {header: sequence}."""
    sequences = {}

    with open(file_path, "r") as file:
        header = None
        seq = []

        for line in file:
            line = line.strip()

            if line.startswith(">"):
                if header is not None:
                    sequences[header] = "".join(seq)

                header = line
                seq = []
            else:
                seq.append(line)

        if header is not None:
            sequences[header] = "".join(seq)

    return sequences


def extract_gene_id(header):
    """Extract gene ID from FASTA header."""
    parts = header.split()

    if len(parts) < 2:
        return None

    return parts[1]  # adjust if needed for your format


def find_longest_mrna(sequences):
    """Keep only the longest isoform per gene."""
    longest_mrna = {}

    for header, seq in sequences.items():
        gene_id = extract_gene_id(header)

        if gene_id is None:
            continue

        if gene_id not in longest_mrna:
            longest_mrna[gene_id] = (header, seq)
        else:
            _, longest_seq = longest_mrna[gene_id]

            if len(seq) > len(longest_seq):
                longest_mrna[gene_id] = (header, seq)

    return longest_mrna


def write_fasta(output_path, longest_mrna):
    """Write FASTA output."""
    with open(output_path, "w") as file:
        for header, seq in longest_mrna.values():
            file.write(f"{header}\n")
            file.write(f"{seq}\n")


# ======================================================
# Main
# ======================================================

if __name__ == "__main__":
    input_file = "sex_genes_appended_protein_reordered.fasta"
    output_file = "longest_mRNAs_protein.fasta"

    sequences = read_fasta(input_file)
    longest_mrna = find_longest_mrna(sequences)

    write_fasta(output_file, longest_mrna)

    print(f"Longest isoforms written to: {output_file}")
```

   ```python
# Calculate density function
def calculate_density(df, bin_size, value_col="name"):
    df["bin"] = (df["start"] // bin_size) * bin_size
    density = df.groupby(["chromosome", "bin"]).size().reset_index(name="density")
```

   ```bash
#!/bin/bash

# Specify the index file
mapping_index="genome_index"

# Create output directories
mkdir -p bam_dir_newgenome
mkdir -p sam_dir_newgenome

# Loop over forward read files
for fwd_input_file in *R.1.fastq
do
    # Extract the sample name by removing .R.1.fastq
    sample_name=$(basename "$fwd_input_file" .R.1.fastq)

    # Define paired file names
    rev_input_file="${sample_name}.R.2.fastq"

    # Define output filenames
    sam_file="${sample_name}.sam"
    bam_file="${sample_name}.bam"

    echo "Processing sample: ${sample_name}"
    echo "Forward read: ${fwd_input_file}"
    echo "Reverse read: ${rev_input_file}"
    echo "Output SAM: ${sam_file}"
    echo "Output BAM: ${bam_file}"

    # Run HISAT2 and convert SAM to BAM
    hisat2 -p 4 --dta --rna-strandness RF \
        -x ${mapping_index} \
        -1 ${fwd_input_file} \
        -2 ${rev_input_file} \
        -S ${sam_file}

    samtools sort -@ 4 -o ${bam_file} ${sam_file}

    rm ${sam_file}  # Remove intermediate SAM file

    # Move output BAM files to the directory
    mv ${bam_file} bam_dir_newgenome/
done
```

```r
library(ape)

tree_files <- list.files(pattern = "\\.treefile$", full.names = TRUE)

dir.create("rooted_trees", showWarnings = FALSE)

pick_one_outgroup <- function(tips) {
    hap <- tips[grepl("Haplomitrium", tips)]
    if (length(hap) > 0) return(hap[1])

    tre <- tips[grepl("Treubia", tips)]
    if (length(tre) > 0) return(tre[1])

    pel <- tips[grepl("Pellia", tips)]
    if (length(pel) > 0) return(pel[1])

    return(NA_character_)
}

for (f in tree_files) {
    cat("Processing:", f, "\n")

    tree <- read.tree(f)
    tips <- tree$tip.label

    og <- pick_one_outgroup(tips)

    if (is.na(og)) {
        cat("  → no Haplomitrium/Treubia/Pellia found; keeping original\n")
        rooted <- tree
    } else {
        cat("  → rooting using outgroup:", og, "\n")

        rooted <- tryCatch(
            root(tree, outgroup = og, resolve.root = TRUE),
            error = function(e) {
                cat("  → rooting failed (", e$message, "); keeping original\n", sep = "")
                return(tree)
            }
        )
    }

    outname <- file.path("rooted_trees", basename(f))
    write.tree(rooted, file = outname)
}
```

def overlap_sum(sorted_intervals, wstart, wend, idx0=0):
    """Calculate total overlap between sorted intervals and a window."""
    total = 0
    n = len(sorted_intervals)
    i = idx0

    while i < n and sorted_intervals[i][1] <= wstart:
        i += 1

    j = i
    while j < n and sorted_intervals[j][0] < wend:
        start, end = sorted_intervals[j]
        overlap_start = max(start, wstart)
        overlap_end = min(end, wend)

        if overlap_end > overlap_start:
            total += overlap_end - overlap_start

        if end >= wend:
            break

        j += 1

    return total, i

"""
Summarize Blasia genes without detectable homologs across genomic regions.

Comparison database:
- Marchantia polymorpha proteome

Outputs:
- region-level homolog summary table
- PAR vs Autosome 2x2 table
- gene-level hit/no-hit table
- bar plot as PNG/PDF
"""

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from Bio import SeqIO
from scipy.stats import chi2_contingency, fisher_exact
from matplotlib.ticker import MaxNLocator


# ======================================================
# Inputs
# ======================================================

blast_path = "blasia_vs_MpTak.tsv"
fasta_path = "sex_genes_appended_protein_reordered_primaryTranscript.fasta"
gff_path = (
    "BlasiaPusilla_Final_annotations_validORF_sexGenesAppended_"
    "renamed_9scaffolds_reordered_myb_modified.gff"
)

SEX_SCAF = "Bp_9"
BP5_SCAF = "Bp_5"
PAR_END = 21_410_000


# ======================================================
# Helper functions
# ======================================================

def extract_attr(attr, key):
    """Extract key=value from GFF attributes."""
    if not isinstance(attr, str):
        return None

    for part in attr.split(";"):
        if part.startswith(key + "="):
            return part.split("=", 1)[1]

    return None


def region(scaf, start):
    """Assign genomic region from scaffold and coordinate."""
    if scaf == BP5_SCAF:
        return "Bp_5"

    if scaf != SEX_SCAF:
        return "Autosome"

    return "PAR" if start < PAR_END else "SDR"


def normalize_id(x):
    """Normalize FASTA/BLAST IDs for matching to GFF IDs."""
    if "|" in x:
        x = x.split("|")[0]

    if "." in x and x.split(".")[-1].isdigit():
        x = ".".join(x.split(".")[:-1])

    if "_" in x and x.split("_")[-1].isdigit():
        x = "_".join(x.split("_")[:-1])

    return x

# 1. Universe of Blasia queries
# ======================================================

# 2. BLASTP hits: presence / absence
# ======================================================

blast_cols = ["qseqid", "sseqid", "evalue", "bitscore", "length", "qlen"]
blast = pd.read_csv(blast_path, sep="\t", names=blast_cols)

hit_ids = set(blast["qseqid"].astype(str))
df["has_hit"] = df["qseqid"].isin(hit_ids)


# ======================================================

# ======================================================

cols = [
    "scaffold",
    "source",
    "type",
    "start",
    "end",
    "score",
    "strand",
    "phase",
    "attr",
]

gff = pd.read_csv(gff_path, sep="\t", comment="#", names=cols)
gff["start"] = gff["start"].astype(int)

# gene -> region
genes = gff[gff["type"] == "gene"].copy()
genes["gene_id"] = genes["attr"].apply(lambda x: extract_attr(x, "ID"))
genes["region"] = genes.apply(lambda r: region(r.scaffold, r.start), axis=1)

gene_map = genes.set_index("gene_id")["region"].to_dict()

# mRNA -> gene
mrna = gff[gff["type"] == "mRNA"].copy()
mrna["mrna_id"] = mrna["attr"].apply(lambda x: extract_attr(x, "ID"))
mrna["gene_id"] = mrna["attr"].apply(lambda x: extract_attr(x, "Parent"))

mrna_gene = mrna.set_index("mrna_id")["gene_id"].to_dict()

    """Assign a genomic region to a query protein ID."""
    if qid in gene_map:
        return gene_map[qid]

        return gene_map[mrna_gene[qid]]

    if normalized in gene_map:
        return gene_map[normalized]

    if normalized in mrna_gene and mrna_gene[normalized] in gene_map:
        return gene_map[mrna_gene[normalized]]

    return np.nan


df["region"] = df["qseqid"].apply(assign_region)

mapped = int(df["region"].notna().sum())
unmapped = int(df["region"].isna().sum())

df["region"] = df["region"].fillna("Unmapped")

print(f"Mapped: {mapped}, Unmapped: {unmapped}")
